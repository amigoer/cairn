{{- /* 卡片样式：完全由 front matter `cardSize` 控制 预设： - "feature" 大卡（顶部 banner 大图 + 标题 / 摘要 / 标签） - "standard" 左侧图卡（左图 + 右内容，默认） - "right" 右侧图卡（左内容 + 右图） - "compact" 无图卡（标题 + 摘要 + 标签） - "note" 单行卡（用于项目等短条目） 未指定时默认：有图 → standard，无图 → compact */ -}} {{- $cardSize := .Params.cardSize | default "" -}} {{- $result := "" -}} {{- if $cardSize -}} {{- $result = $cardSize -}} {{- else -}} {{- /* 默认行为：检测是否有图片 */ -}} {{- $hasImage := false -}} {{- with .Params.cover -}} {{- if or (hasPrefix . "http") (hasPrefix . "//") (hasPrefix . "/") -}} {{- $hasImage = true -}} {{- else -}} {{- with $.Resources.GetMatch . -}}{{ $hasImage = true }}{{- end -}} {{- end -}} {{- end -}} {{- if not $hasImage -}} {{- with .Resources.ByType "image" -}} {{- with index . 0 -}}{{ $hasImage = true }}{{- end -}} {{- end -}} {{- end -}} {{- if $hasImage -}} {{- $result = "standard" -}} {{- else -}} {{- $result = "compact" -}} {{- end -}} {{- end -}} {{ return $result }}
